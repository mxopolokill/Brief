//ecran de redemarrage de la partie 
  function printScreenRestart(){
    var ctx = this.disp.getContext("2d");//ajout du contexte
    ctx.beginPath();//ouverture chemin 
    ctx.rect(this.posx, this.posy, this.sizex, this.sizey);//création rectangle
    ctx.fillStyle = "#222222";// modification remplissage couleur 
    ctx.fill();//remplissage modification 
    ctx.stroke();//fermeture chemin 
   
    var fontSize = this.sizey/10;// taille police d'écriture 
  
    ctx.fillStyle = "white";//modification couleur de remplissage
    ctx.strokeStyle = "white";//modification couleur de fermeture 
    ctx.font = fontSize + "px Arial";//modification police d'écriture 
    ctx.textAlign = "center";//alignement du text au centre
    ctx.fillText("option de redémarrafe",this.posx + this.sizex/2 ,this.posy + this.sizey/4);// ajout du texte de rédemarrage + positionnement
    ctx.fillText("redemarrer avec ce nom",this.posx + this.sizex/2 ,this.posy + this.sizey/2);//ajout du texte du premier choix + positionnement
    ctx.fillText("redemarrer avec un autre nom",this.posx + this.sizex/2 ,this.posy + this.sizey/2 + fontSize*2);//ajout du second choix + positionnement 
    ctx.closePath();//fermeture chemin 
  }
