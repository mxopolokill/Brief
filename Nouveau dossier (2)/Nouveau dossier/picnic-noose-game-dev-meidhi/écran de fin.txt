//écran de Loose 
  function printLoseScreen(){
    var ctx = this.disp.getContext("2d");//ajout du context
    ctx.beginPath();//création chemin 
    ctx.rect(this.posx, this.posy, this.sizex, this.sizey);//création rectangle 
    ctx.fillStyle = "#222222";//couleur de remplissement 
    ctx.fill();//remplissage
    ctx.stroke();//dessine chemin actuelle 

    var fontSize = this.sizey/9;//modification taille écriture 

    ctx.fillStyle = "white";//choix couleur remplissage 
    ctx.font = fontSize + "px Arial";//modification du style d'écriture 
    ctx.textAlign = "center";// alignement du texte au center 
    ctx.fillText("T nul, Tu as perdu !",this.posx + this.sizex/2 ,this.posy + this.sizey/2);//ajout du message de Loose avec sa position
    ctx.fillText("Score : " + this.score ,this.posx + this.sizex/2 ,this.posy + this.sizey/2 + fontSize*2);//ajout du message score avec sa position 
    ctx.closePath();//fermeture chemin 
    sleep(5000).then(()=> {//pause de 5s, 5000ms = 5S
      this.started=3;
      this.StorageScore(this.score);//récupere le score pour l'implémenter dans le Storage score 
    });
  }
